    func (s *TenantService) ListFees(ctx context.Context, tenantID string, claims *auth.Claims) ([]*entity.MandatoryFee, []*entity.VoluntaryFee, error) {
        // Verify tenant access first
        tenant, err := s.GetByID(ctx, tenantID, claims)
        if err != nil {
            return nil, nil, err
        }
        _ = tenant // access verified

        mandatoryFees, err := s.feeRepo.ListMandatoryByTenant(ctx, tenantID)
        if err != nil {
            return nil, nil, fmt.Errorf("failed to list mandatory fees: %w", err)
        }

        voluntaryFees, err := s.feeRepo.ListVoluntaryByTenant(ctx, tenantID)
        if err != nil {
            return nil, nil, fmt.Errorf("failed to list voluntary fees: %w", err)
        }

        if mandatoryFees == nil {
            mandatoryFees = []*entity.MandatoryFee{}
        }
        if voluntaryFees == nil {
            voluntaryFees = []*entity.VoluntaryFee{}
        }

        return mandatoryFees, voluntaryFees, nil
    }

    // CreateMandatoryFee creates a mandatory fee for a tenant after validating access and constraints.
    func (s *TenantService) CreateMandatoryFee(ctx context.Context, tenantID string, amount decimal.Decimal, description string, effectiveDate time.Time, claims *auth.Claims) (*entity.MandatoryFee, error) {
        // Verify tenant access
        tenant, err := s.GetByID(ctx, tenantID, claims)
        if err != nil {
            return nil, err
        }

        // Validate fee
        if err := s.validateFee(amount, tenant.MonthlyFee, effectiveDate); err != nil {
            return nil, err
        }

        fee := &entity.MandatoryFee{
            TenantID:      tenantID,
            Amount:        amount,
            Description:   description,
            EffectiveDate: effectiveDate,
        }

        created, err := s.feeRepo.CreateMandatory(ctx, fee)
        if err != nil {
            return nil, fmt.Errorf("failed to create mandatory fee: %w", err)
        }

        return created, nil
    }

    // CreateVoluntaryFee creates a voluntary contribution for a tenant after validating access and constraints.
    func (s *TenantService) CreateVoluntaryFee(ctx context.Context, tenantID string, amount decimal.Decimal, description string, effectiveDate time.Time, claims *auth.Claims) (*entity.VoluntaryFee, error) {
        // Verify tenant access
        tenant, err := s.GetByID(ctx, tenantID, claims)
        if err != nil {
            return nil, err
