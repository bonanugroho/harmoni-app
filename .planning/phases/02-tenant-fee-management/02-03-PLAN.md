    ```go
    package middleware

    import (
        "log"

        "harmoni-api/internal/infrastructure/auth"

        "github.com/gofiber/fiber/v2"
    )

    // AuditCrossTerritoryAccess logs all API access attempts for audit trail.
    // This middleware runs after the Casbin middleware and logs:
    // - User who made the request (ID, role, territory)
    // - The HTTP method and path
    // - Whether access was granted or denied (based on response status code)
    func AuditCrossTerritoryAccess() fiber.Handler {
        return func(c *fiber.Ctx) error {
            // Process the request first
            err := c.Next()

            // Extract user claims for audit logging
            user := c.Locals("user")
            if user != nil {
                if claims, ok := user.(*auth.Claims); ok {
                    statusCode := c.Response().StatusCode()
                    // Log all access â€” both granted and denied per D-02 auditing requirement
                    log.Printf("AUDIT: user=%s role=%s territory=%s method=%s path=%s status=%d",
                        claims.UserID, claims.Role, claims.TerritoryID,
                        c.Method(), c.Path(), statusCode)

                    // Log specifically for 403 (cross-territory denial)
                    if statusCode == fiber.StatusForbidden {
                        log.Printf("AUDIT:DENY: user=%s role=%s territory=%s attempted=%s %s",
                            claims.UserID, claims.Role, claims.TerritoryID,
                            c.Method(), c.Path())
                    }
                }
            }

            return err
        }
    }
    ```
