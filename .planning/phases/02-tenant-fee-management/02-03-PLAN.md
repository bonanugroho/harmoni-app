    **B) Create a custom resource extractor function in casbin.go:**

    In `casbin.go`, add a new exported function `TenantResourceExtractor` that the TenantHandler can use. This function extracts the first path segment after `/api/` but normalizes `tenants` → `tenant` to match the existing Casbin policy convention where resource names are singular:

    ```go
    // TenantResourceExtractor extracts the resource name from tenant/fee URLs.
    // For /api/tenants/:id/fees, returns "tenant" as the resource (singular, matching policy convention).
    // For /api/fees/:feeId, returns "fee" as the resource.
    func TenantResourceExtractor(c *fiber.Ctx) string {
        path := c.Path()
        // Remove leading slash
        if len(path) > 0 && path[0] == '/' {
            path = path[1:]
        }
        // Remove /api/ prefix
        if len(path) >= 4 && path[:4] == "api/" {
            path = path[4:]
        }
        // Get first segment
        var firstSeg string
        for i, ch := range path {
            if ch == '/' {
                firstSeg = path[:i]
                break
            }
        }
        if firstSeg == "" {
            firstSeg = path
        }
        // Normalize: "tenants" → "tenant", "fees" → "fee"
        switch firstSeg {
        case "tenants":
            return "tenant"
        case "fees":
            return "fee"
        default:
            return firstSeg
        }
    }
    ```

    ```go
    package middleware
    
    import (
        "log"
        "harmoni-api/internal/infrastructure/auth"
        "github.com/gofiber/fiber/v2"
    )
    
    // AuditCrossTerritoryAccess returns middleware that logs all access attempts
    // and flags cross-territory violations. Designed to run before Casbin middleware
    // to capture request context for audit trail.
    func AuditCrossTerritoryAccess() fiber.Handler {
        return func(c *fiber.Ctx) error {
            user := c.Locals("user")

                }
            }
            return c.Next()
        }
    }
    ```
